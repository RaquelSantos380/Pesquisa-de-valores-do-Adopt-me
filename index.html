const historico = []; function buscarPet() { const input = document.getElementById('pesquisa'); const filtro = input.value.trim().toLowerCase(); const lista = document.getElementById('listaPets'); lista.innerHTML = ''; if (filtro && !historico.includes(filtro)) { historico.unshift(filtro); if (historico.length > 10) { historico.pop(); } atualizarHistorico(); } if (!filtro) { // Mostrar todos os pets quando não há filtro pets.forEach(pet => { const item = criarItemPet(pet); lista.appendChild(item); }); return; } const resultados = pets.filter(pet => pet.nome.toLowerCase().includes(filtro) || pet.categoria.toLowerCase().includes(filtro) ); if (resultados.length === 0) { const item = document.createElement('li'); item.textContent = "Nenhum pet encontrado. Tente outro termo."; item.style.color = '#999'; item.style.fontStyle = 'italic'; lista.appendChild(item); } else { resultados.forEach(pet => { const item = criarItemPet(pet); lista.appendChild(item); }); } } function criarItemPet(pet) { const item = document.createElement('li'); // Determinar classe da categoria let categoriaClass = 'categoria-'; if (pet.categoria.toLowerCase().includes('lendario')) { categoriaClass += 'lendario'; } else if (pet.categoria.toLowerCase().includes('raro')) { categoriaClass += 'raro'; } else { categoriaClass += 'comum'; } item.innerHTML = ` ${pet.nome} ${pet.categoria} R$ ${pet.valor.toFixed(2).replace('.', ',')} `; return item; } function atualizarHistorico() { const historicoLista = document.getElementById('historicoPesquisas'); historicoLista.innerHTML = ''; if (historico.length === 0) { const item = document.createElement('li'); item.textContent = "Nenhuma pesquisa realizada ainda."; item.style.color = '#999'; item.style.fontStyle = 'italic'; historicoLista.appendChild(item); return; } historico.forEach(termo => { const item = document.createElement('li'); item.innerHTML = ` 🔍 ${termo} `; historicoLista.appendChild(item); }); } function removerDoHistorico(termo) { const index = historico.indexOf(termo); if (index > -1) { historico.splice(index, 1); atualizarHistorico(); } } // Calculadora let calcInput = ""; function toggleCalculator() { const calculator = document.getElementById("calculator"); const isVisible = calculator.style.display === "block"; calculator.style.display = isVisible ? "none" : "block"; // Animar o botão const button = document.getElementById("toggle-button"); button.style.transform = isVisible ? "scale(1)" : "scale(1.1) rotate(90deg)"; } function appendToCalculator(value) { calcInput += value; document.getElementById("calcInput").value = calcInput; } function clearCalculator() { calcInput = ""; document.getElementById("calcInput").value = ""; } function calculate() { try { // Substituir vírgula por ponto para cálculo const expression = calcInput.replace('×', '*').replace(',', '.'); const result = eval(expression); // Verificar se o resultado é um número válido if (isNaN(result) || !isFinite(result)) { throw new Error('Resultado inválido'); } // Formatar o resultado (máximo 2 casas decimais) const formattedResult = Number.isInteger(result) ? result.toString() : result.toFixed(2).replace('.', ','); document.getElementById("calcInput").value = formattedResult; calcInput = formattedResult.toString().replace(',', '.'); } catch (error) { document.getElementById("calcInput").value = "Erro!"; setTimeout(() => { clearCalculator(); }, 1500); } } // Teclado da calculadora document.addEventListener('keydown', function(event) { const key = event.key; // Verificar se a tecla é um número, operador ou ponto if (/[0-9+\-*/.=]/.test(key)) { event.preventDefault(); if (key === '=' || key === 'Enter') { calculate(); } else if (key === 'Escape' || key === 'Delete') { clearCalculator(); } else { appendToCalculator(key === '*' ? '×' : key); } } // Fechar calculadora com Escape if (key === 'Escape') { const calculator = document.getElementById("calculator"); if (calculator.style.display === "block") { toggleCalculator(); } } }); // Carregar todos os pets inicialmente document.addEventListener('DOMContentLoaded', function() { buscarPet(); atualizarHistorico(); }); // Fechar calculadora ao clicar fora document.addEventListener('click', function(event) { const calculator = document.getElementById("calculator"); const toggleButton = document.getElementById("toggle-button"); if (calculator.style.display === "block" && !calculator.contains(event.target) && !toggleButton.contains(event.target)) { toggleCalculator(); } });
